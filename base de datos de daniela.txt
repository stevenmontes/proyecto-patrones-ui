CREATE PROCEDURE [dbo].[pa_iniciar_sesion] 
@usuario VARCHAR(25),
@clave VARCHAR(25)
AS
	BEGIN TRY
		BEGIN TRANSACTION
			SELECT cedula, primer_nombre, primer_apellido, id_area_funcional 
			FROM templeado 
			WHERE @usuario = usuario
			AND @clave = clave
		COMMIT TRANSACTION
	END TRY
	BEGIN CATCH
	END CATCH
GO
